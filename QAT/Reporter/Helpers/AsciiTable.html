def to_s
  return @ascii if @ascii

  # conver to 2D array
  table  = set_table(content)
  widths = calculate_widths(table)

  # header separator
  table.insert(1, widths.map { |n| '-' * n })

  format = widths.collect { |n| "%-#{n}s" }.join(" | ")

  @ascii ||= table.map do |line|
    sprintf "| #{format} |\n", *line
  end.join
end
